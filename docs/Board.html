def add_chip(col, value)
  return false unless valid_move?(col)

  @board[col] << value
  update_move(value)

  true

    # File lib/board.rb, line 25
def valid_move?(col)
  return false if col.nil? || col.negative?
  return false if @board[col].nil?

  @board[col].length < HEIGHT
end
